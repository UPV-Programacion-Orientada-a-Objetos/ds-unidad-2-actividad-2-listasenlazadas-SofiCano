41 // Constructor que inicializa el rotor con el alfabeto A-Z y espacio
42 RotorDeMapeo() {
43 // Inicializar con A-Z y espacio (27 caracteres)
44 const char alfabeto[] = "ABCDEFGHIJKLMNOPQRSTUVWXYZ ";
45 int longitud = 27;
46
47 NodoRotor* primero = nullptr;
48 NodoRotor* ultimo = nullptr;
49

51 for (int i = 0; i < longitud; i++) {
52 NodoRotor* nuevoNodo = new NodoRotor(alfabeto[i]);
53
54 if (primero == nullptr) {
55 primero = nuevoNodo;
56 ultimo = nuevoNodo;
57 cabeza = primero; // La cabeza apunta a 'A'

59 ultimo->siguiente = nuevoNodo;
60 nuevoNodo->previo = ultimo;
61 ultimo = nuevoNodo;
62 }
63 }
64

66 ultimo->siguiente = primero;
67 primero->previo = ultimo;

70 // Rota el rotor N posiciones (positivo = adelante, negativo = atras)
71 // Mueve el puntero cabeza N posiciones de forma circular
72 void rotar(int n) {
73 if (n > 0) {
74 // Rotar hacia adelante
75 for (int i = 0; i < n; i++) {
76 cabeza = cabeza->siguiente;
77 }
78 } else if (n < 0) {
79 // Rotar hacia atras
80 for (int i = 0; i > n; i--) {
81 cabeza = cabeza->previo;
82 }
83 }

86 // Obtiene el caracter mapeado segun la rotacion actual del rotor
87 // Busca el caracter 'in' y retorna el caracter mapeado
88 char getMapeo(char in) {
89 // Buscar el caracter en el rotor
90 NodoRotor* actual = cabeza;
91 int posicion = 0;
92 bool encontrado = false;
93
94 // Primera vuelta completa para encontrar el caracter
95 NodoRotor* inicio = cabeza;
96 do {
97 if (actual->dato == in) {
98 encontrado = true;
99 break;
100 }
101 posicion++;
102 actual = actual->siguiente;
103 } while (actual != inicio);
104
105 if (!encontrado) {
106 // Si no se encuentra, devolver el mismo caracter
107 return in;
108 }
109
110 // El mapeo es: la posicion del caracter desde la cabeza actual
111 // determina que caracter del alfabeto original representa
112 return cabeza->dato;
113 }
114
115 // Destructor que libera toda la memoria de la lista circular
116 ~RotorDeMapeo() {
117 if (cabeza == nullptr) return;
118
119 NodoRotor* actual = cabeza->siguiente;
120 NodoRotor* temp;
121
122 // Romper el circulo
123 cabeza->siguiente = nullptr;
124
125 while (actual != nullptr) {
126 temp = actual;
127 actual = actual->siguiente;
128 delete temp;
129 }
130 delete cabeza;
131 }

248// Elimina espacios en blanco al inicio y final de una cadena
249void trim(char* str) {
250 int inicio = 0;
251 int fin = strlen(str) - 1;
252
253 // Eliminar espacios al inicio
254 while (str[inicio] == ' ' || str[inicio] == '\t' ||
255 str[inicio] == '\n' || str[inicio] == '\r') {
256 inicio++;
257 }
258
259 // Eliminar espacios al final
260 while (fin >= inicio && (str[fin] == ' ' || str[fin] == '\t' ||
261 str[fin] == '\n' || str[fin] == '\r')) {
262 fin--;
263 }
264
265 // Mover la cadena al inicio
266 int i;
267 for (i = 0; inicio <= fin; i++, inicio++) {
268 str[i] = str[inicio];
269 }
270 str[i] = '\0';
271}

273// Parsea una linea de trama y crea el objeto correspondiente
274// Formato esperado: "L,X" o "M,N"
275// Retorna puntero a TramaBase o nullptr si hay error
276TramaBase* parsearTrama(char* linea) {
277 // Limpiar la linea
278 trim(linea);
279
280 // Verificar longitud minima
281 if (strlen(linea) < 3) {
282 return nullptr;
283 }
284
285 char tipo = linea[0];
286
287 // Verificar que tiene coma
288 if (linea[1] != ',') {
289 return nullptr;
290 }
291
292 // Obtener el valor despues de la coma
293 char* valor = linea + 2;
294
295 if (tipo == 'L' || tipo == 'l') {
296 // Trama LOAD
297 char caracter = valor[0];
298
299 // Manejar caso especial de "Space"
300 if (strcmp(valor, "Space") == 0 || strcmp(valor, "space") == 0 ||
301 strcmp(valor, "SPACE") == 0) {
302 caracter = ' ';
303 }
304
305 return new TramaLoad(caracter);
306 }
307 else if (tipo == 'M' || tipo == 'm') {
308 // Trama MAP
309 int rotacion = atoi(valor);
310 return new TramaMap(rotacion);
311 }
312
313 return nullptr;
314}
